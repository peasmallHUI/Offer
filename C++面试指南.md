    - 二叉树高度
        ```c++
        template<typename T>
        int BinaryTree<T>::Height(BinaryTreeNode<T> *t) const
        {
            if (!t) return 0;
            int hl = Height(t->LeftChild);
            int hr = Height(t->RightChild);
            if (hl > hr) return ++hl;
            else return ++hr;
        }
        ```
